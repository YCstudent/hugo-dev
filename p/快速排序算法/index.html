const int N = 1000010;
int q[N];
int n;

void quick_sort(int q[], int l, int r)
{
    if (l >= r) return;

    int i = l - 1, j = r + 1, x = q[l + r >> 1];
    while (i < j)
    {
        do i ++; while(q[i] < x);
        do j --; while(q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j), quick_sort(q, j + 1, r);
}

int main()
{
    scanf("%d", &n);

    for (int i = 0; i < n; i ++) scanf("%d", &q[i]);

    quick_sort(q, 0, n - 1);

    for (int i = 0; i < n; i++) printf("%d ", q[i]);
    return 0;
}

const int N = 1000010;

int q[N];
int n, k;

int quick_sort(int l, int r, int k)
{
    if (l == r) return q[l];

    int i = l - 1, j = r + 1, x = q[l + r >> 1];

    while (i < j)
    {
        do i++;while (q[i] < x);
        do j--;while (q[j] > x);
        if (i < j) swap (q[i], q[j]);
    }
   int sl = j - l + 1;
   if (k <= sl) return quick_sort(l, j, k);

   else return quick_sort(j + 1, r, k - sl);
}

{
    cin >> n >> k;
